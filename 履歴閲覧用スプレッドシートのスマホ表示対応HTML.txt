<script>
  let rawHistory = [];
  let myChart = null;

  window.onload = loadData;

  function switchTab(tabId) {
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.content').forEach(c => c.classList.remove('active'));
    document.querySelector(`.tab[onclick="switchTab('${tabId}')"]`).classList.add('active');
    document.getElementById(tabId).classList.add('active');
  }

  // ★修正: エラーハンドラ(withFailureHandler)を追加し、フリーズを防止
  function loadData() {
    document.getElementById('loading').style.display = 'block';
    google.script.run
      .withSuccessHandler(renderData)
      .withFailureHandler(function(error) {
        document.getElementById('loading').style.display = 'none';
        alert("データ取得エラーが発生しました。\n詳細: " + error.message);
      })
      .getDataForWeb();
  }

  function renderData(rawData) {
    document.getElementById('loading').style.display = 'none';
    
    // ★修正: GASから送られたJSON文字列を正しくJavaScriptのデータに変換
    let data;
    try {
      data = JSON.parse(rawData);
    } catch(e) {
      data = rawData;
    }

    rawHistory = data.history || [];

    // ★修正: 存在しない system.mode ではなく、データ取得完了時刻を表示
    const nowTime = new Date().toLocaleTimeString('ja-JP', {hour: '2-digit', minute: '2-digit'});
    document.getElementById('sysInfo').textContent = `最終更新: ${nowTime}`;

    // カード生成
    const cardsDiv = document.getElementById('cards-container');
    cardsDiv.innerHTML = '';
    
    if (data.log && data.log.length > 0) {
      data.log.forEach(row => {
        if (!row || row.length < 2) return;
        const ts = new Date(row[0]).toLocaleString('ja-JP', {month:'2-digit', day:'2-digit', hour:'2-digit', minute:'2-digit'});
        const isErr = row[11] !== 'NORMAL' || row[12] === 'ACT_ERR';
        const statusClass = isErr ? 'status-ERROR' : 'status-NORMAL';
        const battClass = (row[9] > 0 && row[9] <= 11.8) ? 'alert' : '';
        
        const html = `
          <div class="card ${statusClass}">
            <div class="card-header"><div class="field-id">${row[1]}</div><div class="timestamp">${ts}</div></div>
            <div class="data-grid">
              <div class="data-item">
                <span class="label">水位</span>
                <div class="value">${Number(row[2]).toFixed(2)}cm</div>
                <div class="level-bar-bg"><div class="level-bar-fill" style="width: ${row[3] ? Number(row[3]) : 0}%"></div></div>
              </div>
              <div class="data-item">
                <span class="label">状態</span>
                <div class="value">${row[8]}</div>
                <span class="label">指令: ${row[6]} / 結果: ${row[7] || "-"}</span>
              </div>
              <div class="data-item">
                <span class="label">バッテリー</span>
                <div class="value" style="${battClass ? 'color:#c62828' : ''}">${Number(row[9]).toFixed(1)}V</div>
              </div>
              <div class="data-item">
                <span class="label">水門Pico電圧</span>
                <div class="value" style="${(row[5] > 0 && row[5] < 2.6) ? 'color:#c62828' : ''}">${Number(row[5]).toFixed(2)}V</div>
              </div>
              <div class="data-item">
                <span class="label">判定</span>
                <div class="value" style="${isErr ? 'color:#c62828' : ''}">${isErr ? '異常' : '正常'}</div>
              </div>
            </div>
          </div>`;
        cardsDiv.insertAdjacentHTML('beforeend', html);
      });
    } else {
      cardsDiv.innerHTML = '<div style="text-align:center; color:#888; padding:20px;">データがありません</div>';
    }

    // 履歴生成
    const histDiv = document.getElementById('history-container');
    histDiv.innerHTML = '';
    if (data.history && data.history.length > 0) {
      data.history.forEach(row => {
        if (!row || row.length < 2) return;
        const ts = new Date(row[0]).toLocaleString('ja-JP', {month:'2-digit', day:'2-digit', hour:'2-digit', minute:'2-digit'});
        const actClass = row[8] === 'OPEN' ? 'act-OPEN' : (row[8] === 'CLOSE' ? 'act-CLOSE' : '');
        const html = `<div class="history-item"><div class="hist-left"><div class="hist-id">${row[1]}</div><div class="hist-time">${ts}</div></div><div class="hist-right"><div class="hist-action ${actClass}">${row[8] || '-'}</div></div></div>`;
        histDiv.insertAdjacentHTML('beforeend', html);
      });
    }

    // グラフ用セレクトボックス
    const select = document.getElementById('fieldSelect');
    select.innerHTML = '';
    if (data.history) {
      const uniqueIds = [...new Set(data.history.map(item => item[1]))].sort();
      uniqueIds.forEach(id => {
        const option = document.createElement('option');
        option.value = id;
        option.text = id;
        select.appendChild(option);
      });
      if (uniqueIds.length > 0) updateChart();
    }
  }

  function updateChart() {
    const selectedId = document.getElementById('fieldSelect').value;
    if (!selectedId) return;
    
    // データをフィルタリングして古い順に
    const filteredData = rawHistory.filter(row => row[1] === selectedId).reverse();

    const labels = [];
    const waterLevels = [];
    const pointColors = [];
    const pointStyles = [];
    const pointRadii = [];
    const gateStatuses = [];

    filteredData.forEach(row => {
      const d = new Date(row[0]);
      labels.push(`${d.getHours()}:${String(d.getMinutes()).padStart(2, '0')}`);
      waterLevels.push(row[2]); // 水位
      
      const status = row[8]; // Gate_Status
      gateStatuses.push(status);

      if (status === 'OPEN') {
        pointColors.push('#2E7D32'); // 緑
        pointStyles.push('triangle');
        pointRadii.push(6); // 大きく目立たせる
      } else if (status === 'CLOSE') {
        pointColors.push('#C62828'); // 赤
        pointStyles.push('rect');
        pointRadii.push(6); // 大きく目立たせる
      } else {
        pointColors.push('rgba(33, 150, 243, 0.5)');
        pointStyles.push('circle');
        pointRadii.push(0); // 通常時は点を描画しない（線のみ）
      }
    });

    const ctx = document.getElementById('myChart').getContext('2d');

    if (myChart) myChart.destroy();

    myChart = new Chart(ctx, {
      type: 'line',
      data: {
        labels: labels,
        datasets: [{
          label: '水位 (V)',
          data: waterLevels,
          borderColor: '#2196F3',
          backgroundColor: 'rgba(33, 150, 243, 0.1)',
          borderWidth: 2,
          pointBackgroundColor: pointColors,
          pointBorderColor: pointColors,
          pointStyle: pointStyles,
          pointRadius: pointRadii,
          tension: 0.1,
          fill: true
        }]
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
        interaction: { mode: 'index', intersect: false },
        plugins: {
          tooltip: {
            callbacks: {
              afterLabel: function(context) {
                // 水門操作があった場合のみツールチップに表示
                const idx = context.dataIndex;
                const st = gateStatuses[idx];
                if (st === 'OPEN' || st === 'CLOSE') {
                  return `操作: ${st}`;
                }
                return null;
              }
            }
          }
        },
        scales: {
          y: {
            beginAtZero: true,
            title: { display: true, text: '水位 (V)' },
            max: 3.5
          }
        }
      }
    });
  }
</script>
